create table Cliente(
	cpf int not null primary key,
    nome varchar(30) not null,
    telefone varchar(15)
);


create table Funcionario(
	cpf int not null primary key,
    nome varchar(30) not null,
    telefone varchar(15),
    endereco varchar(40) not null,    
	especialidade varchar(30) not null
    
);

create table Servico(
    	id int auto_increment primary key,           
		horario int not null,    
		calendario varchar(15) not null,
        preco double not null
);

alter table Servico add column id_cliente int not null;
alter table Servico add column id_func int not null;

alter table Servico add foreign key (id_func) references Funcionario(cpf);
alter table Servico add foreign key (id_cliente) references cliente(cpf);


create table Login(	
    login varchar(40) not null primary key,
    senha varchar(40) not null
);
